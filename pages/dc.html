const int motor = 11;
            const int pt = A0;

            byte vmotor = 255;
            void setup() {
            Serial.begin(9600);
            }


            int p = 0;

            void loop() {
            p = analogRead(pt);
            if ((p>=0) && (p<512)) {
                vmotor = 127;
            } else {
                vmotor = 0;
            }
            
            analogWrite(motor, vmotor);
            }


Требования:

  • 1 Источник питания с напряжением 6 и током 1.
  • 1 Потенциометр.
  • 1 n-канальный МОП-транзистор (полевой транзистор с МОП-Структурой)
  • 1 NPN-транзистор (биополярный)
  • 1 диод
  • 1 двигатель постоянного тока
  • 1 резистор на 1kΩ
  • 1 резистор на 10kΩ

Результат: Мотор крутится и его скорость можно регулировать через потенциометр.
