#include  <KeyPad.h>
                #include  <LiquidCrystal.h>
                
                LiquidCrystal lcd(1, 0, 13, 12, 11, 10);
                
                const byte numRows= 4; 
                const byte numCols= 4; 
                
                char keymap[numRows][numCols]= 
                {
                {'1', '2', '3', 'A'}, 
                {'4', '5', '6', 'B'}, 
                {'7', '8', '9', 'C'},
                {'*', '0', '#', 'D'}
                };
                
                byte rowPins[numRows] = {9,8,7,6};
                byte colPins[numCols]= {5,4,3,2}; 
                
                Keypad myKeypad = Keypad(makeKeymap(keymap), rowPins, colPins, numRows, numCols);
                
                void setup() {
                   lcd.begin(16, 1);
                }
                
                void loop() {
                  
                  
                    char keypressed = myKeypad.getKey();
                      if (keypressed != NO_KEY) {
                      if (keypressed == '#') {
                        lcd.clear();
                        lcd.print("Calling");
                        delay(260);
                        lcd.print(".");
                        delay(260);
                        lcd.print(".");
                        delay(260);
                        lcd.print(".");
                        delay(300);
                        lcd.clear();
                        lcd.print("No money.");
                        delay(1000);
                        lcd.clear();
                      } else {
                        if (keypressed == '*') {
                        lcd.print("-");
                        } else {
                          lcd.print(keypressed);
                        }
                      }
                      
                      
                    }
                }


Требования:

  • 1 Клавиатура 4 * 4.
  • 1 ЖК-экран.
  • 1 Стабилитрон с напряжением 5.10kV.
  • 1 Ползунковый переключатель.
  • 1 Разъем с 8 выводами.

Результат: Можно набирать номер и звонить по нему.
