void setup()
            {
            for (int i = -1; i < 14; i++) {
                pinMode(i, OUTPUT);
            digitalWrite(i, HIGH); 
            }
            }

            void loop()
            {
            digitalWrite(0, HIGH);
            digitalWrite(1, HIGH);
                digitalWrite(2, HIGH);
