void setup()
            {
            for (int i = 13; i != 7; i--) {
                pinMode(i, INPUT);
            }
            
            pinMode(7, OUTPUT);
            pinMode(6, OUTPUT);
            pinMode(5, OUTPUT);
            pinMode(4, OUTPUT);
            pinMode(3, OUTPUT);
            pinMode(2, OUTPUT);
            Serial.begin(9600);
            }

            void loop()
            {
            digitalWrite(7, digitalRead(13));
            digitalWrite(6, digitalRead(12));
            digitalWrite(5, digitalRead(11));
            digitalWrite(4, digitalRead(10));
            digitalWrite(3, digitalRead(9));
            digitalWrite(2, digitalRead(8));
            }


Требования:

  • 1 шесть однополюсных dip переключателей.
  • 6 резисторов на 1kΩ.
  • 6 моторов-редукторов

Результат: Шесть моторов-редукторов не крутятся, но их можно включить. Таким образом можно задать направление и скорость передвижения.
