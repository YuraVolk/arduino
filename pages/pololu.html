#include <Servo.h> 

        Servo motor;
        
        #define PIN_POTENTIOMETER 2
        #define PIN_PWM 9
        int val; 
         int encoder0PinA = 3;
         int encoder0PinB = 4;
         int encoder0Pos = 0;
         int encoder0PinALast = LOW;
         int n = LOW;
        
        void setup() {
          pinMode (encoder0PinA,INPUT);
           pinMode (encoder0PinB,INPUT);
           Serial.begin (9600);
          motor.attach(PIN_PWM);
        
        }
        
        void loop() {
          int potValue = analogRead(PIN_POTENTIOMETER);
          int speed = map(potValue, 0, 1023, 0, 255);
          motor.write(speed);
         n = digitalRead(encoder0PinA);
           if ((encoder0PinALast == LOW) && (n == HIGH)) {
             if (digitalRead(encoder0PinB) == LOW) {
               encoder0Pos--;
             } else {
               encoder0Pos++;
             }
             Serial.print (speed);
             Serial.print ("/");
           } 
           encoder0PinALast = n;
          delay(10);
        }


Требования:

  • 1 Pololu
  • 1 источник питания с напряжением 12 и током 12
  • 1 потенциометр
  • 1 двигатель постоянного тока с энкодером на 1621 об/мин

Результат: Вы сможете регулировать скорость мотора. Кол-Во тестов - 2.
