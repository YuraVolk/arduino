void setup()
            {
            pinMode(7, INPUT);
            pinMode(6, INPUT);
            pinMode(5, INPUT);
            pinMode(4, INPUT);
            pinMode(3, OUTPUT);
            pinMode(2, OUTPUT);
            pinMode(1, OUTPUT);
            Serial.begin(9600);
            }

            void loop()
            {
            if (digitalRead(7) == 0) {
                RGB_color(map(digitalRead(6), 0, 1, 0, 255),
                map(digitalRead(5), 0, 1, 0, 255),
                map(digitalRead(4), 0, 1, 0, 255));
            } else {
                RGB_color(map(digitalRead(4), 0, 1, 0, 255),
                map(digitalRead(5), 0, 1, 0, 255),
                map(digitalRead(6), 0, 1, 0, 255));
            }
            }

            void RGB_color(int red_light_value, int green_light_value, int blue_light_value)
            {
            Serial.println(red_light_value);
            
            Serial.println(green_light_value);
            Serial.println(blue_light_value);
            
            analogWrite(3, red_light_value);
            analogWrite(2, green_light_value);
            analogWrite(1, blue_light_value);
            }


Требования:

  • 1 четыре двухполюсных dip переключателей.
  • 4 резистора на 1kΩ.
  • 3 резистора на 220Ω.
  • 1 RGB-светодиод.

Результат: Можно настраивать цвет светодиода.
